# 1、axios

# (1) 封装

使用 axios 时只应该提供method、url和data，所以按照需求多级封装一下。

  • baseUrl
  • headers
  • status
  • methods
  • timeout
  • interceptors

# (2) 简单实现

send(config){
    return new Promise(resolve => {
        const {url = '', method = 'get', data = {}} = config;
        const xhr = new XMLHttpRequest();
        xhr.open(method, url, true);
        xhr.onload = function() {
            resolve(xhr.responseText);
        };
        xhr.send(data);
    })
}
